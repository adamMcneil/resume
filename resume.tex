% TimeFrame

% #3 Length of time
\newcommand{\timeFrame}[3] {
  \if\relax\detokenize{#2}\relax
    #1
    % {\space\space\space\space\space#1}
  \else
    #1 - #2
  \fi
  \newline
  \scriptsize \textit{#3}
}

\newcommand{\entry}[6] {
  % #1 Employer/Header
  \textbf{#1}

  \if\relax\detokenize{#2}\relax
  \else
       - \textsc{#2}
  \fi
  % #3 Location
  \if\relax\detokenize{#3}\relax
  \else
      - \textsc{#3}
  \fi
  % #4 URL
  \if\relax\detokenize{#4}\relax
  \else
      - \href{https://#4}{#4}
  \fi
  % #4 Description
  \newline #5
  % #5 Technologies used
  \if\relax\detokenize{#6}\relax
  \else
      \newline \textsc{Technology:} #6
  \fi
  \\
}
