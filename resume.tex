\newcommand{\timeFrame}[3] {
  \if\relax\detokenize{#2}\relax
    #1
    % {\space\space\space\space\space#1}
  \else
    #1 - #2
  \fi
  \newline
  \scriptsize \textit{#3}
}

  \fi

  \else
      - \href{https://#4}{#4}
  \fi

  \newline #5
  % #5 Technologies used
  \if\relax\detokenize{#6}\relax
  \else
      \newline \textsc{Technology:} #6
